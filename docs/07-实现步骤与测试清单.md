```cpp
    UPyManimCamera* GetCamera();

private:
    bool bInitialized = false;
    UPROPERTY()
    APyManimSceneActor* SceneActor;
    UPROPERTY()
    UPyManimCamera* Camera;
    UPROPERTY()
    TArray<UPyManimMobject*> Mobjects;
};
```

```cpp
// PyManimScene.cpp
#include "PyManimScene.h"
#include "PyManimCamera.h"
#include "Engine/World.h"
#include "EngineUtils.h"

void UPyManimScene::Initialize(int32 Width, int32 Height)
{
    if (bInitialized) return;

    UWorld* World = GEditor ? GEditor->GetEditorWorldContext().World() : nullptr;
    if (!World) return;

    FTransform SpawnTransform(FRotator::ZeroRotator, FVector::ZeroVector);
    SceneActor = World->SpawnActor<APyManimSceneActor>(SpawnTransform);

    if (SceneActor)
    {
        Camera = NewObject<UPyManimCamera>(this);
        bInitialized = true;
    }
}

bool UPyManimScene::IsInitialized() const
{
    return bInitialized;
}

UPyManimMobject* UPyManimScene::CreateSphere(float Radius)
{
    if (!bInitialized || !SceneActor) return nullptr;

    UPyManimMobject* NewObj = NewObject<UPyManimMobject>(this);
    if (NewObj)
    {
        NewObj->InitializeAsSphere(SceneActor, Radius);
        Mobjects.Add(NewObj);
    }
    return NewObj;
}

UPyManimCamera* UPyManimScene::GetCamera()
{
    return Camera;
}
```

```cpp
#include "CoreMinimal.h"
#include "PyManimMobject.generated.h"

class UMaterialInterface;

class UPyManimMobject : public UObject
{
    GENERATED_BODY()

public:

    UFUNCTION(BlueprintCallable, Category = "PyManim|Mobject")
    void SetVisibility(bool bVisible);

    UFUNCTION(BlueprintCallable, Category = "PyManim|Mobject")
    void SetScale(const FVector& Scale);

private:
    void CreateProceduralMesh(APyManimSceneActor* Owner);

    FString MobjectName;
    bool bVisible = true;
    FLinearColor CurrentColor = FLinearColor::White;

```

```cpp
// PyManimMobject.cpp
#include "PyManimMobject.h"
#include "PyManimSceneActor.h"
#include "Components/StaticMeshComponent.h"
#include "Engine/StaticMesh.h"
#include "Materials/MaterialInstanceDynamic.h"
#include "ProceduralMeshComponent.h"
#include "KismetProceduralMeshLibrary.h"

void UPyManimMobject::InitializeAsSphere(APyManimSceneActor* Owner, float Radius)
{
    if (!Owner) return;

    CreateProceduralMesh(Owner);
}

void UPyManimMobject::CreateProceduralMesh(APyManimSceneActor* Owner)
{
    UWorld* World = Owner->GetWorld();
    if (!World) return;

    // 用 StaticMeshActor 作为载体（比 ProceduralMesh 更稳定）
    FActorSpawnParameters Params;
    Params.Name = FName(*FString::Printf(TEXT("Mobject_%d"), GetUniqueID()));
    AActor* Actor = World->SpawnActor<AActor>(Params);

    if (!Actor) return;
    InternalActor = Actor;

    // 添加 Mesh 组件
    UStaticMeshComponent* MeshComp = NewObject<UStaticMeshComponent>(Actor);
    MeshComp->RegisterComponent();
    Actor->SetRootComponent(MeshComp);

    // 加载默认球体 Mesh
    UStaticMesh* SphereMesh = LoadObject<UStaticMesh>(nullptr, TEXT("/Engine/BasicShapes/Sphere.Sphere"));
    if (SphereMesh)
    {
        MeshComp->SetStaticMesh(SphereMesh);
    }

    // 设置材质为动态实例以便改颜色
    UMaterialInterface* BaseMaterial = LoadObject<UMaterialInterface>(
        nullptr, TEXT("/Engine/BasicShapes/BasicShapeMaterial.BasicShapeMaterial"));

    if (BaseMaterial && MeshComp)
    {
        MaterialInstance = UMaterialInstanceDynamic::Create(BaseMaterial, this);
        MeshComp->SetMaterial(0, MaterialInstance.Get());
    }

    MeshComponent = MeshComp;
}

void UPyManimMobject::SetLocation(const FVector& Location)
{
    if (InternalActor.IsValid())
    {
        InternalActor->SetActorLocation(Location);
    }
}

FVector UPyManimMobject::GetLocation() const
{
    if (InternalActor.IsValid())
        return InternalActor->GetActorLocation();
    return FVector::ZeroVector;
}

void UPyManimMobject::SetColor(const FLinearColor& Color)
{
    CurrentColor = Color;
    if (MaterialInstance.IsValid())
    {
        MaterialInstance->SetVectorParameterValue(TEXT("BaseColor"), Color);
    }
}

FLinearColor UPyManimMobject::GetColor() const
{
    return CurrentColor;
}

void UPyManimMobject::SetVisibility(bool bInVisible)
{
    bVisible = bInVisible;
    if (MeshComponent.IsValid())
    {
        MeshComponent->SetVisibility(bVisible);
    }
}

void UPyManimMobject::SetScale(const FVector& Scale)
{
    if (InternalActor.IsValid())
    {
        InternalActor->SetActorScale3D(Scale);
    }
}
```

### Step 2.1 — 实现 UPyManimCamera

### Step 2.2 — 实现 UPyManimAnimation 基类

**目标**：动画基类框架（Duration / Easing / Tick / IsFinished）。

**操作**：新建 `Anim/PyManimAnimation.h/.cpp`：

```cpp
UCLASS(BlueprintType)
class UPyManimAnimation : public UObject
{
    GENERATED_BODY()
public:
    UFUNCTION(BlueprintCallable)
    void SetDuration(float InDuration) { Duration = InDuration; }

    UFUNCTION(BlueprintCallable)
    float GetDuration() const { return Duration; }

    UFUNCTION(BlueprintCallable)
    void SetEasing(const FString& Type) { EasingType = Type; }

    UFUNCTION(BlueprintCallable)
    bool IsFinished() const { return Elapsed >= Duration; }

    virtual void Tick(float DeltaTime, float Progress) {}

    void Advance(float DeltaTime)
    {
        Elapsed += DeltaTime;
        float Progress = FMath::Clamp(Elapsed / Duration, 0.0f, 1.0f);
        float EasedProgress = ApplyEasing(EasingType, Progress);
        Tick(DeltaTime, EasedProgress);
    }

    static float ApplyEasing(const FString& Type, float t);

protected:
    float Duration = 1.0f;
    float Elapsed = 0.0f;
    FString EasingType = "linear";
};
```

**验证测试**：

```python

assert not anim.is_finished()
```

```cpp
public:
```

### Step 2.9 — 实现 Scene.Tick() 与 Play() 驱动

**目标**：`scene.play(animation)` 后自动驱动动画逐帧推进。

**核心逻辑**：

```cpp
void UPyManimScene::Play(UPyManimAnimation* Animation)
{
    if (!Animation || !bInitialized) return;
    ActiveAnimations.Add(Animation);
    Animation->Advance(0);  // Reset
}

void UPyManimScene::Tick(float DeltaTime)
{
    if (!bInitialized) return;

    for (int32 i = ActiveAnimations.Num() - 1; i >= 0; --i)
    {
        auto* Anim = ActiveAnimations[i];
        Anim->Advance(DeltaTime);
        if (Anim->IsFinished())
        {
            ActiveAnimations.RemoveAt(i);
        }
    }
}
```

**注意**：UE 编辑器模式下需要注册一个 Tick 回调来持续调用 `Scene::Tick()`。可通过 `FTicker` 或自定义 `FEditorTick` 实现。

**验证测试**：

```python

    scene.tick(1.0/30.0)  # 30fps

# 或者用定时器自动 tick（如果已实现）
```
